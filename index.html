// 🔊 緊張感を高めたロングアラーム音 function playStrongAlarmSound() { const duration = 2.5; // 音の長さを2.5秒に延長 const now = audioCtx.currentTime; // ベースとなる不協和音（サイレン風のうねり） let osc1 = audioCtx.createOscillator(); let gain1 = audioCtx.createGain(); osc1.type = 'sawtooth'; // より鋭い鋸歯状波に変更 osc1.frequency.setValueAtTime(150, now); // 時間経過で音程を少し下げて緊迫感を出す osc1.frequency.linearRampToValueAtTime(120, now + duration); osc1.connect(gain1); gain1.connect(audioCtx.destination); gain1.gain.setValueAtTime(0.3, now); gain1.gain.exponentialRampToValueAtTime(0.001, now + duration); let osc2 = audioCtx.createOscillator(); let gain2 = audioCtx.createGain(); osc2.type = 'sawtooth'; osc2.frequency.setValueAtTime(156, now); // 周波数をずらして激しい「うねり」を作る osc2.frequency.linearRampToValueAtTime(126, now + duration); osc2.connect(gain2); gain2.connect(audioCtx.destination); gain2.gain.setValueAtTime(0.3, now); gain2.gain.exponentialRampToValueAtTime(0.001, now + duration); // 金属的な不快感（アラート感）を強調する高音を追加 let osc3 = audioCtx.createOscillator(); let gain3 = audioCtx.createGain(); osc3.type = 'square'; // 矩形波でパルス感を出す osc3.frequency.setValueAtTime(600, now); osc3.connect(gain3); gain3.connect(audioCtx.destination); gain3.gain.setValueAtTime(0.15, now); // 高音は途中で途切れるようにして不規則な緊張感を演出 gain3.gain.setValueAtTime(0.15, now + 0.5); gain3.gain.exponentialRampToValueAtTime(0.001, now + 1.5); // すべての音を開始・停止 osc1.start(now); osc2.start(now); osc3.start(now); osc1.stop(now + duration); osc2.stop(now + duration); osc3.stop(now + 1.5); }
